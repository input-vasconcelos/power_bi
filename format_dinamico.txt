    [LM total R$],

LM total R$ = 
VAR vMesAtual = [Total R$]

VAR vMesAnterior = CALCULATE(
    [Total R$],
    PREVIOUSMONTH(dCalendario[Data])
)
VAR vPorcentagem = DIVIDE(
    vMesAtual - 
    vMesAnterior, vMesAnterior,
    0)

    RETURN
    vPorcentagem
